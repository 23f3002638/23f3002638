yaml
# docker-compose.yml
version: '3.8'
services:
  backend:
    build: ./backend
    ports:
      - "5000:5000"
    environment:
      - MONGO_URI=mongodb://mongo:27017/terminus
    depends_on:
      - mongo
  mongo:
    image: mongo:latest
    volumes:
      - mongo-data:/data/db
  frontend:
    build: ./frontend
    ports:
      - "3000:3000"
volumes:
  mongo-data:
bash
# Spin up the whole stack
docker-compose up -d
